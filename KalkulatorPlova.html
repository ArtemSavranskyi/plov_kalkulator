            <button id="calculateBtn"

                    class="w-full bg-blue-600 text-white font-bold py-3 px-4 rounded-xl shadow-lg hover:bg-blue-700 transition duration-300 transform hover:scale-105 focus:outline-none focus:ring-4 focus:ring-blue-300">

                Розрахувати

            </button>

        </div>


        <div id="resultBox" class="mt-8 hidden">

            <div class="bg-gray-100 p-6 rounded-2xl border border-gray-300 text-center">

                <h3 class="text-xl font-bold text-gray-800">Результат:</h3>

                <p id="resultText" class="mt-2 text-2xl font-extrabold text-blue-600 animate-pulse">

                    <!-- Результат буде тут -->

                </p>

    <script>

        document.addEventListener('DOMContentLoaded', () => {

            const ageInput = document.getElementById('age');

            const heightInput = document.getElementById('height');

            const weightInput = document.getElementById('weight');

            const calculateBtn = document.getElementById('calculateBtn');

            const resultBox = document.getElementById('resultBox');

            const resultText = document.getElementById('resultText');


            calculateBtn.addEventListener('click', () => {

                // Отримуємо значення з полів та перетворюємо їх на числа

                const age = parseFloat(ageInput.value);

                const height = parseFloat(heightInput.value);

                const weight = parseFloat(weightInput.value);


                // Проста валідація вхідних даних

                if (isNaN(age) || isNaN(height) || isNaN(weight) || age <= 0 || height <= 0 || weight <= 0) {

                    resultText.textContent = "Будь ласка, введіть коректні позитивні числа.";

                    resultBox.classList.remove('hidden');

                    resultBox.classList.add('bg-red-100', 'border-red-400');

                    resultText.classList.remove('text-blue-600', 'animate-pulse');

                    resultText.classList.add('text-red-700');

                    return;

                }


                // Вигадана формула для розрахунку порції плова

                // Це просто приклад, щоб калькулятор працював :)

                let basePlov = 250; // Базова порція в грамах

                

                // Додаємо вагу

                const weightFactor = (weight - 70) * 3;

                

                // Додаємо зріст

                const heightFactor = (height - 170) * 1.5;

                

                // Додаємо вік

                const ageFactor = (age - 30) * 0.8;

                

                let plovAmount = basePlov + weightFactor + heightFactor + ageFactor;


                // Мінімальна порція 150г, щоб не було від'ємних значень

                plovAmount = Math.max(150, plovAmount);


                // Округлюємо до найближчого цілого числа

                const roundedAmount = Math.round(plovAmount);

                

                // Відображаємо результат

                resultBox.classList.remove('bg-red-100', 'border-red-400', 'hidden');

                resultBox.classList.add('bg-gray-100', 'border-gray-300');

                resultText.classList.remove('text-red-700');

                resultText.classList.add('text-blue-600', 'animate-pulse');

                resultText.textContent = `${roundedAmount} грам плову. Смачного!`;

            });

        });

    </script>
